#include <stdio.h>


typedef struct s{
    int id,at,bt,ct,tat,wt;
} s;

void swap(struct s *a,struct s *b)
{
    struct s temp;
    temp=*a;
    *a=*b;
    *b=temp;
}
void sort(struct s *arr,int n)
{
    for(int i=0;i<n-1;i++)
    {
        for(int j=0;j<n-i-1;j++)
        {
            if(arr[j].at>arr[j+1].at)
            {
                swap(&arr[j],&arr[j+1]);
            }
        }
    }
}

void table(s *a,int n)
{
    printf("pid\tat\tbt\tct\ttat\twt\n");
    for(int i=0;i<n;i++)
    {
        printf("%d\t%d\t%d\t%d\t%d\t%d\n ",a[i].id,a[i].at,a[i].bt,a[i].ct,a[i].tat,a[i].wt);
    }
}
int main()
{
    int n;
    printf("enter the no. of process:");
    scanf("%d",&n);
    
    s arr[n];
    int a,b;
    printf("enter arrival time and burst time of processes:\n");
    for(int i=0;i<n;i++)
    {
        scanf("%d %d",&a,&b);
        arr[i].id=i+1;
        arr[i].at=a;
        arr[i].bt=b;
    }
    table(arr,n);
    printf("\n");
    sort(arr,n);
    
    
    int sum=0;
    for(int i=0;i<n;i++)
    {
        if(sum>=arr[i].at){
            arr[i].ct=arr[i].bt+sum;
            sum=arr[i].ct;
        }
        else
        {
            arr[i].ct=arr[i].bt+arr[i].at;
            sum=arr[i].ct;
        }
    }
    
    
    for(int i=0;i<n;i++)
    {
        arr[i].tat = arr[i].ct - arr[i].at;
        arr[i].wt = arr[i].tat - arr[i].bt;
    }
    table(arr,n);
    
    int avg_wt=0;
    int avg_tat=0;
    
    for(int i=0;i<n;i++)
    {
        avg_tat += arr[i].tat;
        avg_wt+= arr[i].wt;
    }
    printf("average waiting time : %d\n",avg_wt);
    printf("average tat : %d",avg_tat);
    return 0;
}
